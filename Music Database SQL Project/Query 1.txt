/*Query 1 – query used for first insight */SELECT a.title album,                SUM(i.UnitPrice*i.Quantity)   FROM Album a   JOIN track t    ON a.albumid = t.albumid    JOIN Invoiceline i   ON i.Invoicelineid = t.trackid   JOIN invoice   ON invoice.invoiceid = i.invoiceid   JOIN customer c    ON c.customerid = invoice.customerid   GROUP BY 1   HAVING c.country = 'USA'   ORDER BY 2 DESC    LIMIT 10  /*Query 2 – query used for Second insight */SELECT c.state state,                  g.name genre, c.country country, COUNT(*) AS numb_times  FROM customer c  JOIN invoice   ON c.customerid = invoice.customerid   JOIN InvoiceLine i   ON i.invoiceid = invoice.invoiceid  JOIN track t   ON t.trackid = i.TrackId  JOIN genre g   ON g.genreid = t.genreid  WHERE c.state IS NOT NULL AND g.name = 'Rock'  GROUP BY 1,2  ORDER BY numb_times DESC  LIMIT 10 /*Query 3 – query used for third insight */SELECT m.name medianame,  COUNT(*) as numb_of_times  FROM MediaType m  JOIN track t   ON t.MediaTypeId = m.mediatypeid  JOIN genre g  ON g.GenreId = t.GenreId  GROUP BY 1 ORDER BY numb_of_times DESC   /*Query 4 – query used for forth insight */  SELECT t.name track,                   c.Country Country,                   SUM(I.quantity*i.UnitPrice)  FROM track t   JOIN InvoiceLine i  ON t.trackid = i.TrackId  JOIN Invoice  ON Invoice.InvoiceId = i.InvoiceId  JOIN Customer c  ON c.CustomerId = Invoice.CustomerId  WHERE country = 'USA'  GROUP BY 1,2  ORDER BY 3 DESC   LIMit 10
